import {
  Controller,
  Get,
  Post,
  Body,
  Patch,
  Param,
  Delete,
  UseGuards,
  Req,
  Request,
  SetMetadata,
} from '@nestjs/common';
import { UserService } from './user.service';
import { CreateUserDto } from './dto/create-user.dto';
import { UpdateUserDto } from './dto/update-user.dto';
import { JwtAuthGuard } from 'src/auth/guards/jwt-auth/jwt-auth.guard';
import { Role } from 'src/auth/enums/role.enum';
import { Roles } from 'src/auth/decorator/role.decorator';
import { RolesGuard } from 'src/auth/guards/roles/roles.guard';

@Controller('user')
export class UserController {
  constructor(private readonly userService: UserService) {}

  @Post()
  create(@Body() createUserDto: CreateUserDto) {
    return this.userService.create(createUserDto);
  }

  // Protect this route with JWT authentication
  // This guard uses the AuthGuard from Passport.js with the 'jwt' strategy
  @UseGuards(JwtAuthGuard)
  @Get('profile')
  async getProfile(@Request() req) {
    console.log(req.user);
    // The user object will be available in the request after authentication
    return this.userService.findOne(req.user.id);
  }

  @Patch(':id')
  update(@Param('id') id: string, @Body() updateUserDto: UpdateUserDto) {
    return this.userService.update(+id, updateUserDto);
  }

  //say we allow only admin user to do this
  //@SetMetadata('role', [Role.ADMIN])
  //use custom decorator
  //accessible to only admin or editor
  @Roles(Role.ADMIN, Role.EDITOR)
  // @UseGuards(RolesGuard)   ? HAS BEEN ENABLED GLOBALLY CHECK AUTH MODULE
  // @UseGuards(JwtAuthGuard)
  @Delete(':id')
  remove(@Param('id') id: string) {
    return this.userService.remove(+id);
  }
}

import { User } from 'src/entities/user.entity';

@Injectable()
export class UserService {
  //inject user repository
  constructor(@InjectRepository(User) private UserRepo: Repository<User>) {}
  async create(createUserDto: CreateUserDto) {
    const user = await this.UserRepo.create(createUserDto);
    //insert the user into the database
    //used create because of the before insert hook, check user.entity.ts
    return await this.UserRepo.save(user);
  }

  async updateHashedRefreshToken(userId: number, hashedRefreshToken: string) {
    return await this.UserRepo.update({ id: userId }, { hashedRefreshToken });
  }

  findAll() {
    return `This action returns all user`;
  }

  //find by email
  async findByEmail(email: string) {
    const user = await this.UserRepo.findOne({
      where: { email },
    });
    return user;
  }

  async findOne(id: number) {
    return await this.UserRepo.findOne({
      where: { id },
      select: [
        'id',
        'firstName',
        'lastName',

        'hashedRefreshToken',
        'role',
      ],
    });
